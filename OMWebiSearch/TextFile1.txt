private void DoClean(DCOMWSearchProfile profile)
{
    foreach (var p in profile.SearchParams)
    {
        if (p.Value.ValueType == EOMWDataType.DateTime)
        {
            var val = p.Value as OMWValueDateTime;
            val.Value = DateTime.Now;
        }
    }
}
